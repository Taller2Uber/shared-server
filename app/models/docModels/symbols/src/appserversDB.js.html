106 /** @name getServerInfo
107 * @function createServer
108 * @memberof appserverDB
109 * @author Gustavo Adrian Gimenez
110 * @param userId Id del appserver del cual se quiere obtener la informacion
111 */
112 appserverDB.getServerInfo = function( response, userId ){
113   client = new pg.Client({connectionString: db.url, ssl:true});
114   var respuestaJson = {};
115   if( db.connectClient( client, response ) ){
116       logger.info("Se conecto a la base de datos correctamente");
117       var results = [];
118       client.query('SELECT * FROM appservers WHERE id = $1', [userId], (err, res) =>{
119         if( err )
120           logger.error('Error en la query');
121         else {
122             res.rows.forEach(row =>{
123             results.push(row);
124             });
125             if( results.length <= 0 ){
126               logger.info('Server inexistente');
127               response.status(404).json(respuesta.addError(respuestaJson, 404, 'Server inexistente'));
128             } else {
129               logger.info('Informacion del server');
130               respuestaJson = respuesta.addDescription(respuestaJson, 'Informacion del server');
131               response.status(200).json(respuesta.addResult(respuestaJson, results[0]));
132             }
133           }
134       });
135     }else{
136       response.status(500).json(respuesta.addError(respuestaJson, 500, 'Unexpected error'));
137     }

141 /** @name updateServerInfo
142 * @function updateServerInfo
143 * @memberof appserverDB
144 * @author Gustavo Adrian Gimenez
145 * @param userId Id del appserver del cual se quiere modificar la informacion
146 */
147 appserverDB.updateServerInfo = function( response, request, userId ){
148   var respuestaJson = {};
149   client = new pg.Client({connectionString: db.url, ssl:true});
150     if( db.connectClient( client, response ) ){
151       client.query('SELECT _ref FROM appservers WHERE id = $1',[userId], (err, res) =>{
152         if(err){
153           logger.info('ERROR: ' + err);
154           return false;
155         }
156         else{
157           res.rows.forEach(row =>{
158             ref = JSON.parse(JSON.stringify(row, null, 2));
159           if( ref._ref === request.body._ref ){
160             var jsonSV = {"name":request.body.name };
161             var newRef = refCheck.generate(jsonSV);
162             client.query('UPDATE appservers SET name = $1, _ref = $2 WHERE id = $3',
163                 [ request.body.name, newRef, userId ], (err, res) =>{
164               if( err ){
165                 logger.info('No existe el recurso solicitado');
166                 response.status(404).json(respuesta.addError(respuestaJson, 404, 'No existe el recurso solicitado'));
167               } else {
168                 logger.info('Actualización de información del server conectado');
169                 respuestaJson = respuesta.addDescription(respuestaJson, 'Modificacion correcta');
170                 jsonSV._ref = newRef;
171                 response.status(200).json(respuesta.addResult(respuestaJson, jsonSV));
172               }
173           })
174         }else{
175           logger.info('Ref incorrecto');
176           response.status(409).json(respuesta.addError(respuestaJson, 409, 'Conflicto en el update. Mal valor de ref'));
177         }});
178       }
179     });
180     }else{
181       logger.info('Unexpected error');
182       response.status(500).send(respuesta.addError(respuestaJson, 500, 'Unexpected error'));
183     }
184 }

187 /** @name renewToken
188 * @function renewToken
189 * @memberof appserverDB
190 * @author Gustavo Adrian Gimenez
191 * @param newToken token nuevo par ael appserver
192 * @param oldRef codigo de hash viejo del server
193 * @param userId id del server al cual se le va a modificar el Token
194 */
195 appserverDB.renewToken = function( response, newToken, oldRef, userId ){
196   var respuestaJson = {};
197   client = new pg.Client({connectionString: db.url, ssl:true});
198     if( db.connectClient( client, response ) ){
199       client.query('SELECT * FROM appservers WHERE id = $1',[userId], (err, res) =>{
200         if(err){
201           logger.info('ERROR: ' + err);
202           return false;
203         }
204         else{
205           res.rows.forEach(row =>{
206             appserver = JSON.parse(JSON.stringify(row, null, 2));
207           if( appserver._ref === oldRef ){
208             var newRef = refCheck.generate(appserver);
209             client.query('UPDATE appservers SET token = $1, _ref = $2 WHERE id = $3',
210                 [ newToken, newRef, userId ], (err, res) =>{
211               if( err ){
212                 logger.info('No existe el recurso solicitado');
213                 response.status(404).json(respuesta.addError(respuestaJson, 404, 'No existe el recurso solicitado'));
214               } else {
215                 logger.info('Actualización de token del server conectado');
216                 respuestaJson = respuesta.addDescription(respuestaJson, 'Modificacion de token correcta');
217                 appserver._ref = newRef;
218                 appserver.token = newToken;
219                 response.status(200).json(respuesta.addResult(respuestaJson, appserver));
220               }
221           })
222         }else{
223           logger.info('Ref incorrecto');
224           response.status(409).json(respuesta.addError(respuestaJson, 409, 'Conflicto en el update. Mal valor de ref'));
225         }});
226       }
227     });
228     }else{
229       logger.info('Unexpected error');
230       response.status(500).send(respuesta.addError(respuestaJson, 500, 'Unexpected error'));
231     }
