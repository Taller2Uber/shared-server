 41 /** @name createUser
 42 * @function createUser
 43 * @memberof appusers
 44 * @author Gustavo Adrian Gimenez
 45 * @param request debe contener todos los datos necesarios para dar de alta un usuario.
 46 */
 47 appusers.createUser = function( response, req ){
 48   var respuestaJson = {};
 49   client = new Client({connectionString: db.url, ssl:true});
 50   if( db.connectClient( client, response ) ){
 51     if( !req.body.username || !req.body.password || !req.body.firstname || !req.body.type || !req.body.lastname || !req.body.country || !req.body.email || !req.body.birthdate ){
 52       logger.error('Incumplimiento de precondiciones (parámetros faltantes)');
 53       response.status(400).json(respuesta.addError(respuestaJson, 400, 'Incumplimiento de precondiciones'));
 54     }else{
 55       var jsonUser = {"username": req.body.username, "password": req.body.password, "firstname": req.body.firstname, "lastname": req.body.lastname,
 56                       "type": req.body.type, "email": req.body.email, "birthdate": req.body.birthdate, "country": req.body.country };
 57       var ref =  refCheck.generate( jsonUser );
 58       client.query('INSERT INTO users (username, password, firstname, lastname, type, email, birthdate, country, _ref) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)',
 59                     [req.body.username, req.body.password, req.body.firstname, req.body.lastname, req.body.type, req.body.email, req.body.birthdate, req.body.country, ref],(err, res) => {
 60       if(err){
 61         console.log(err, res)
 62       }else{
 63         respuestaJson = respuesta.addDescription(respuestaJson, 'Alta correcta');
 64         jsonUser._ref = ref;
 65         response.status(201).json(respuesta.addResult(respuestaJson, jsonUser));
 66       }})}
 67 
 68   }else{
 69     response.status(500).json(respuesta.addError(respuestaJson, 500, 'Unexpected error'));
 70   }
 71 }

118 /** @name getUser
119 * @function getUser
120 * @memberof appusers
121 * @author Gustavo Adrian Gimenez
122 * @param userID id del usuario del cual se quiere obtener la informacion
123 */
124 appusers.getUser = function( response, userId ){
125   var respuestaJson = {};
126   var results = [];
127   client = new Client({connectionString: db.url, ssl:true});
128   if( db.connectClient( client, response ) ){
129     client.query('SELECT * FROM users WHERE id = $1', [userId], (err, res)=>{
130       if(err || res.rows.length <= 0){
131         response.status(404).json(respuesta.addError(respuestaJson, 404, 'User inexistente'));
132       }else{
133         res.rows.forEach(row =>{
134           results.push(row);
135         })
136         respuestaJson = respuesta.addDescription(respuestaJson, 'Informacion del usuario');
137         response.status(200).json(respuesta.addResult(respuestaJson, results));
138       }
139     });
140   }else{
141     response.status(500).json(respuesta.addError(respuestaJson, 500, 'Unexpected error'));
142   }
