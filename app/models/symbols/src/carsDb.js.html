 11 carsDB.getAllCarsFromId = function( response, userId ){
 12   var respuestaJson = {};
 13   var results = [];
 14   client = new Client({connectionString: db.url, ssl:true});
 15   if( db.connectClient( client, response ) ){
 16     client.query('SELECT * FROM cars WHERE  owner = $1', [userId], (err, res) =>{
 17       if(err || res.length <=0 ){
 18         respuestaJson = respuesta.addError(respuestaJson, 404, 'El recurso solicitado no existe');
 19         response.status(404).json(respuestaJson);
 20       }else{
 21         res.rows.forEach(row =>{
 22           results.push(row);
 23         })
 24       }
 25       respuestaJson = respuesta.addDescription(respuestaJson, 'Informacion de autos de un usuario');
 26       respuestaJson = respuesta.addResult(respuestaJson, results);
 27       response.status(200).json(respuestaJson);
 28     });
 29   }else{
 30     respuestaJson = respuesta.addError(respuestaJson, 500, 'Unexpected error');
 31     response.status(500).json(respuestaJson);
 32   }
