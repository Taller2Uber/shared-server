 57 buDB.prototype.checkAuth = function( response, request , session){
 58   if ( !request.body.username || !request.body.password ){
 59     logger.info('Informacion faltante');
 60     response.status(500).send('Informacion faltante');
 61     session = request.session;
 62     session.authenticated = false;
 63     return;
 64   }
 65 
 66   client = new Client({connectionString: db.url, ssl:true});
 67     if( db.connectClient( client, response ) ){
 68       logger.info('Se conecto a la base de datos correctamente.')
 69       var results = [];
 70       client.query('SELECT * FROM businessusers WHERE username = $1 AND password = $2',[request.body.username, request.body.password], (err, res) =>{
 71         if(err)
 72           response.send('Error en la query');
 73         else{
 74             res.rows.forEach(row =>{
 75             results.push(row);
 76             });
 77           checkLogin(results, response, request, session);
 78         }
 79       });

 83 function checkLogin(results, response, request, session){
 84   if( results.length <= 0 ){
 85     logger.info('Usuario y password incorrectos');
 86     var respuestaJson = {};
 87     respuestaJson = respuesta.addError(respuestaJson, 500, 'Username y password incorrectos');
 88     response.status(500).json(respuestaJson);
 89     client.end();
 90   } else {
 91     logger.info('Inicio de sesion correcta');
 92     session = request.session;
 93     session.authenticated = true;
 94     session.role = results[0]['role'];
 95     logger.info('Role: ' + session.role );
 96     session.userId = results[0]['id'];
 97     response.status(200).send('Inicio de sesion correcta')
 98     client.end();
 99   }

133 buDB.prototype.updateInfo = function( response, request, userId ){
134   var respuestaJson = {};
135   client = new Client({connectionString: db.url, ssl:true});
136     if( db.connectClient( client, response ) ){
137       client.query('SELECT _ref FROM businessusers WHERE id = $1',[userId], (err, res) =>{
138         if(err){
139           logger.info('ERROR: ' + err);
140           return false;
141         }
142         else{
143           res.rows.forEach(row =>{
144             ref = JSON.parse(JSON.stringify(row, null, 2));
145           if( ref._ref === request.body._ref ){
146             var jsonBU = {"username":request.body.username, "password":request.body.password, "name":request.body.name, "surname":request.body.surname, "role":request.body.role };
147             var newRef = refCheck.generate(jsonBU);
148             client.query('UPDATE businessusers SET name = $1, username = $2, password = $3, surname = $4, role = $5, _ref = $6 WHERE id = $7',
149                 [ request.body.name, request.body.username, request.body.password, request.body.surname, request.body.role ,newRef, userId ], (err, res) =>{
150               if( err ){
151                 logger.info('No existe el recurso solicitado');
152                 respuestaJson = respuesta.addError(respuestaJson, 404, 'No existe el recurso solicitado');
153                 response.status(404).json(respuestaJson);
154               } else {
155                 logger.info('Actualizaci√≥n de informaci√≥n del usuario de negocio conectado');
156                 respuestaJson = respuesta.addDescription(respuestaJson, 'Modificacion correcta');
157                 jsonBU._ref = newRef;
158                 respuestaJson = respuesta.addResult(respuestaJson, jsonBU);
159                 response.status(200).json(respuestaJson);
160               }
161           })
162         }else{
163           logger.info('Ref incorrecto');
164           respuestaJson = respuesta.addError(respuestaJson, 409, 'Conflicto en el update. Mal valor de ref');
165           response.status(409).json(respuestaJson);
166         }});
167       }
168     });
169     }else{
170       logger.info('Unexpected error');
171       respuestaJson = respuesta.addError(respuestaJson, 500, 'Unexpected error');
172       response.status(500).send(respuestaJson);
173 }}
